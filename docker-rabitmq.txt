//imagen
docker pull rabbitmq

//contenedor
docker run -d --hostname mi-rabbit-server --name rabbit-facu rabbitmq:3

//servicio para ver las queves

docker network connect microservicenet rabbit-facu-web

//verificar que se agregó
docker network inspect microservicenet
